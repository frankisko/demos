Para crear un mapa se necesita:

  • Cargar el javascript de maps.
  • Crear el mapa en un elemento de la página, indicando las opciones que contendrá el mapa.

Map Options

<script src="https://maps.googleapis.com/maps/api/js?sensor=false"></script>
<script>
var map; 
function initialize() {
	//opciones del mapa
	var mapOptions = {
		zoom: 14,
		center: new google.maps.LatLng(20.674346933909842,-103.38696956634521),
		mapTypeId: google.maps.MapTypeId.ROADMAP
	};
	//creacion de mapa en el elemento html con las opciones del mapa
	map = new google.maps.Map(document.getElementById('map_canvas'), mapOptions);
}
	
//mostrar mapa al cargar la página 
google.maps.event.addDomListener(window, 'load', initialize);
</script>
